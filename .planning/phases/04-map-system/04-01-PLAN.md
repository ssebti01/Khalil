# Plan 04-01: Map System Infrastructure

## Goal

Introduce a data-driven map system by:
1. Defining a structured map config format in `src/config/maps.js`
2. Creating `src/systems/MapLoader.js` that renders backgrounds and spawns physics bodies from config
3. Refactoring `GameScene` so `_drawArena()` and `_createWalls()` are delegated to `MapLoader`, with `mapId` accepted from init data

The default "Stadium" map must produce **identical** visuals and physics to the current hardcoded arena.

## Context

**Relevant source (read before editing):**
- `src/scenes/GameScene.js` — `_drawArena()` (lines 80-136) and `_createWalls()` (lines 138-167) are the two methods being replaced. `init()` (lines 13-21) must gain `mapId`. `create()` (lines 23-78) must call `MapLoader` instead of the two private methods.
- `src/config/constants.js` — `GAME_WIDTH`, `GAME_HEIGHT`, `GOAL`, `PHYSICS` are used throughout arena drawing and wall creation. `MapLoader` imports these same constants.

**Physics layer boundary:**
- Goal posts, crossbars, and goal sensors are match logic. They stay in `GameScene._createGoals()` and `GameScene._setupGoalSensors()`. Do NOT move them into `MapLoader`.
- `MapLoader.createObstacles()` owns: floor, ceiling, outer walls (above goal opening), goal-back walls, and any map-specific obstacle bodies defined in `mapConfig.obstacles`.

**Phaser 3 / Matter.js API:**
- Graphics: `scene.add.graphics()`, `.fillGradientStyle()`, `.fillStyle()`, `.fillRect()`, `.lineStyle()`, `.lineBetween()`, `.strokeRect()`, `.strokeCircle()`
- Physics bodies: `scene.matter.add.rectangle(x, y, w, h, options)` — `options` accepts `{ isStatic, friction, restitution, label, angle }`
- Angled bodies: pass `angle` in radians to `matter.add.rectangle`

**Details — stadium config object (exact values, referencing constants):**

```javascript
import { GAME_WIDTH, GAME_HEIGHT, GOAL, PHYSICS } from './constants.js';

export const MAPS = [
  {
    id: 'stadium',
    name: 'Stadium',
    floorRestitution: 0.2,
    floorFriction: 0.3,
    background: {
      type: 'gradient',
      skyColors: [0x0d1b2a, 0x1b3a6b],   // top-left, bottom-right for fillGradientStyle
      pitchColor: 0x2d7a3a,
      pitchStripeAlt: 0x347a42,
      crowdColor: 0x0a1520,
      netBgColor: 0x001a33,
      netLineColor: 0xaaccff,
      floorLineColor: 0x3a9a4a,
      markingColor: 0x3a9a4a,
    },
    obstacles: [],
    windForce: null,
    specialZones: [],
  },
];

export function getMap(id) {
  const m = MAPS.find(m => m.id === id);
  if (!m) throw new Error(`Unknown map: ${id}`);
  return m;
}

export function getMaps() {
  return MAPS;
}
```

The `background` sub-object intentionally stores all color constants so `MapLoader` never hardcodes colors — it reads them from config.

### Task 2: Create `src/systems/MapLoader.js`

**File:** `src/systems/MapLoader.js`

**Action:** Create the file from scratch. Implement two exported functions. Extract all drawing and physics-body logic that currently lives in `GameScene._drawArena()` and `GameScene._createWalls()` into these functions — producing identical output for the `stadium` config.

**Details — `drawBackground(scene, mapConfig)`:**

Port `_drawArena()` exactly, replacing all hardcoded color literals with reads from `mapConfig.background.*`. Crowd silhouette loop uses `Phaser.Math.Between` — this is fine (decorative, not gameplay). Goal post line drawing (the `bg.lineBetween(...)` calls for posts and crossbars) is also decorative and stays here.

```javascript
import { GAME_WIDTH, GAME_HEIGHT, GOAL } from '../config/constants.js';

export function drawBackground(scene, mapConfig) {
  const { background: bg_cfg } = mapConfig;
  const g = scene.add.graphics();

  // Sky gradient
  const [tl, br] = bg_cfg.skyColors;
  g.fillGradientStyle(tl, tl, br, br, 1);
  g.fillRect(0, 0, GAME_WIDTH, GAME_HEIGHT);

  // Crowd silhouettes
  g.fillStyle(bg_cfg.crowdColor, 0.8);
  for (let i = 0; i < 60; i++) {
    const x = i * 22 + 5;
    const h = Phaser.Math.Between(30, 60);
    g.fillRect(x, GAME_HEIGHT - 80 - h - 8, 16, h);
  }

  // Pitch
  g.fillStyle(bg_cfg.pitchColor);
  g.fillRect(0, GAME_HEIGHT - 80, GAME_WIDTH, 80);

  // Pitch markings
  g.lineStyle(2, bg_cfg.markingColor, 0.6);
  g.strokeRect(80, GAME_HEIGHT - 80, GAME_WIDTH - 160, 80);
  g.lineBetween(GAME_WIDTH / 2, GAME_HEIGHT - 80, GAME_WIDTH / 2, GAME_HEIGHT);
  g.strokeCircle(GAME_WIDTH / 2, GAME_HEIGHT - 80, 60);

  // Grass stripes
  for (let i = 0; i < 10; i++) {
    g.fillStyle(i % 2 === 0 ? bg_cfg.pitchColor : bg_cfg.pitchStripeAlt, 0.5);
    g.fillRect(80 + i * ((GAME_WIDTH - 160) / 10), GAME_HEIGHT - 80, (GAME_WIDTH - 160) / 10, 80);
  }

  // Floor highlight line
  g.lineStyle(3, bg_cfg.floorLineColor);
  g.lineBetween(0, GAME_HEIGHT - 80, GAME_WIDTH, GAME_HEIGHT - 80);

  // Net backgrounds
  g.fillStyle(bg_cfg.netBgColor, 0.5);
  g.fillRect(0, GAME_HEIGHT - 80 - GOAL.height, 60, GOAL.height);
  g.fillRect(GAME_WIDTH - 60, GAME_HEIGHT - 80 - GOAL.height, 60, GOAL.height);

  // Net grid
  g.lineStyle(1, bg_cfg.netLineColor, 0.3);
  for (let r = 0; r < GOAL.height; r += 18) {
    g.lineBetween(0, GAME_HEIGHT - 80 - GOAL.height + r, 60, GAME_HEIGHT - 80 - GOAL.height + r);
    g.lineBetween(GAME_WIDTH - 60, GAME_HEIGHT - 80 - GOAL.height + r, GAME_WIDTH, GAME_HEIGHT - 80 - GOAL.height + r);
  }
  for (let c = 0; c < 60; c += 18) {
    g.lineBetween(c, GAME_HEIGHT - 80 - GOAL.height, c, GAME_HEIGHT - 80);
    g.lineBetween(GAME_WIDTH - 60 + c, GAME_HEIGHT - 80 - GOAL.height, GAME_WIDTH - 60 + c, GAME_HEIGHT - 80);
  }

  // Decorative goal posts (visual only — physics posts are in GameScene)
  g.lineStyle(6, 0xdddddd);
  g.lineBetween(60, GAME_HEIGHT - 80 - GOAL.height, 60, GAME_HEIGHT - 80);
  g.lineBetween(GAME_WIDTH - 60, GAME_HEIGHT - 80 - GOAL.height, GAME_WIDTH - 60, GAME_HEIGHT - 80);
  g.lineBetween(60, GAME_HEIGHT - 80 - GOAL.height, 0, GAME_HEIGHT - 80 - GOAL.height);
  g.lineBetween(GAME_WIDTH - 60, GAME_HEIGHT - 80 - GOAL.height, GAME_WIDTH, GAME_HEIGHT - 80 - GOAL.height);
}
```

**Details — `createObstacles(scene, mapConfig)`:**

Port `_createWalls()` base bodies exactly, then iterate `mapConfig.obstacles`:

```javascript
export function createObstacles(scene, mapConfig) {
  const w = GAME_WIDTH, h = GAME_HEIGHT;
  const thick = 40;
  const crossbarY = h - 80 - GOAL.height;

  // --- Base walls (always created, override physics from config) ---
  scene.matter.add.rectangle(w / 2, h - 80 + thick / 2, w, thick, {
    isStatic: true, friction: mapConfig.floorFriction,
    restitution: mapConfig.floorRestitution, label: 'floor',
  });
  scene.matter.add.rectangle(w / 2, -thick / 2, w, thick, {
    isStatic: true, restitution: 0.5, label: 'ceiling',
  });

  const leftWallH = crossbarY;
  scene.matter.add.rectangle(-thick / 2, leftWallH / 2, thick, leftWallH, {
    isStatic: true, restitution: 0.3, label: 'wall',
  });
  scene.matter.add.rectangle(-thick / 2, h - 80 - GOAL.height / 2, thick, GOAL.height, {
    isStatic: true, restitution: 0.4, label: 'goalback',
  });
  scene.matter.add.rectangle(w + thick / 2, leftWallH / 2, thick, leftWallH, {
    isStatic: true, restitution: 0.3, label: 'wall',
  });
  scene.matter.add.rectangle(w + thick / 2, h - 80 - GOAL.height / 2, thick, GOAL.height, {
    isStatic: true, restitution: 0.4, label: 'goalback',
  });

  // --- Map-defined obstacles ---
  const g = scene.add.graphics();

  for (const obs of mapConfig.obstacles) {
    // Draw visual
    if (obs.visual) {
      g.fillStyle(obs.visual.color, obs.visual.alpha ?? 1);
      if (obs.type === 'circle') {
        g.fillCircle(obs.x, obs.y, obs.r);
      } else {
        // box / ramp / platform — all rectangular visuals
        // For angled ramps: draw at center, apply rotation via a temp container if needed.
        // Simple approach: fillRect at top-left position derived from center coords.
        g.fillRect(obs.x - obs.w / 2, obs.y - obs.h / 2, obs.w, obs.h);
      }
    }

    // Spawn physics body
    const bodyOpts = {
      isStatic: true,
      friction: obs.friction ?? 0.05,
      restitution: obs.restitution ?? 0.3,
      label: obs.label ?? 'obstacle',
    };
    if (obs.angle) bodyOpts.angle = obs.angle;

    if (obs.type === 'circle') {
      scene.matter.add.circle(obs.x, obs.y, obs.r, bodyOpts);
    } else {
      scene.matter.add.rectangle(obs.x, obs.y, obs.w, obs.h, bodyOpts);
    }
  }
}
```
